339 
340  // Set the values to which the boundaries of the sliding window is compered to the boundaries of the data cubes
341  int valX = m_shiftXTest*(1-trainFlag)+m_shiftXTrain*trainFlag;
342  int valY = m_shiftYTest*(1-trainFlag)+m_shiftYTrain*trainFlag;
343  int valZ = m_shiftZTest*(1-trainFlag)+m_shiftZTrain*trainFlag;
344  int compareValueY = m_yWidthTest*(1-trainFlag)+m_yWidthTrain*trainFlag;
345  int compareValueX = m_xWidthTest*(1-trainFlag)+m_xWidthTrain*trainFlag;
346  int compareValueZ = m_zWidthTest*(1-trainFlag)+m_zWidthTrain*trainFlag;
347 
348 
349  // If the (starting point in Y dimention + shift + Y dimention of the patach) less than the boundary in Y
350  if ((*startYWidth)+valY+m_patchY <= compareValueY){
351 
352  // Shift the starting value
353  (*startYWidth) += valY;
354 
355  } else {
356 
357  // Start at 0 in the Y dimention of the cube
358  (*startYWidth) = 0;
359 
360  // If the (starting point in X dimention + shift + X dimention of the patach) less than the boundary in X
361  if ((*startXWidth)+valX+m_patchX <= compareValueX){
362 
363  // Shift the starting value
364  (*startXWidth) += valX;
365  } else {
366 
367  // Start at 0 in the X dimention of the cube
368  (*startXWidth) = 0;
369 
370  // If the (starting point in Z dimention + shift + Z dimention of the patach) less than the boundary in Z
371  if ((*startDepth)+valZ+m_patchZ <= compareValueZ){
372 
373  // Start at 0 in the Z dimention of the cube
374  (*startDepth) += valZ;
375 
376  // If reached this state then reached the end of the data cube, set the counter to the max number of iterations
377  } else {
378 
379  (*numbIterations) = ii;
380  }
381  }
382  }
383 
384  // The last example makes the data to shift and brings inconsistency, do not update
385  if (ii == (*numbIterations)-1) (*numbIterations)--;

434 
435  // The procefure follows: 10 log (|x_t|/|x_t-x_p|) where x_t - target, x_p - predicted
436 
437  double qDenom = 0.0;
438  double qNum = 0.0;
439 
440  double clean, noise;
441  int maskCounter = 0;
442 
443  int counter = 0, startZ = 0, startX = 0, startY = 0;
444 
445 
446  // Create two cubes which hold the predicted and the target data
447  vector< vector < vector < double > > > cubeClean;
448  vector< vector < vector < double > > > cubeNoisy;
449 
450  // Allocate the memory for the cubes
451  cubeClean.resize(m_zWidthTest);
452  cubeNoisy.resize(m_zWidthTest);
453  for (int ii = 0; ii < m_zWidthTest; ++ii) {
454  cubeClean[ii].resize(m_xWidthTest);
455  cubeNoisy[ii].resize(m_xWidthTest);
456  for (int jj = 0; jj < m_xWidthTest; ++jj){
457  cubeClean[ii][jj].resize(m_yWidthTest);
458  cubeNoisy[ii][jj].resize(m_yWidthTest);
459  }
460  }
461 
462  // Go through the test examples and copy the data from training arrays to the data cubes, scaling by the number of overlapping patches
463  for (int ii = 0; ii < m_numbItTest; ++ii){
464  for (int jj = startZ + m_augmentZ; jj<startZ + m_patchZ - m_augmentZ; ++jj){
465  for (int kk = startX + m_augmentX; kk < startX + m_patchX - m_augmentX; ++kk){
466  for (int ff = startY + m_augmentY; ff < startY + m_patchY - m_augmentY; ++ff){
467  counter = ii*m_patchX*m_patchY*m_patchZ + (jj - startZ)*m_patchX*m_patchY + (kk - startX)*m_patchY + (ff - startY);
468  cubeNoisy[jj][kk][ff] += dataNoize[counter]/m_coefCube[jj][kk][ff];
469  cubeClean[jj][kk][ff] += dataClean[counter]/m_coefCube[jj][kk][ff];
470  }
471  }
472  }
473  calculateBounds(0,&startY,&startX,&startZ,ii,&m_numbItTest);
474  }
475 
476 
477  // The loop to bring the data back to its original domain, by applying inverse scaling
478  for (int jj = 0; jj < m_zWidthTest; ++jj) {
479  for (int ii = 0; ii < m_xWidthTest; ++ii) {
480  for (int kk = 0; kk < m_yWidthTest; ++kk) {
481 
482  //Apply the inverse scaling
483  if (m_inputScale == -2.0) {
484  clean = cubeClean[jj][ii][kk]*(m_max-m_min)+m_min;
485  noise = cubeNoisy[jj][ii][kk]*(m_max-m_min)+m_min;
486  } else {
487  clean = cubeClean[jj][ii][kk]/m_inputScale;
488  noise = cubeNoisy[jj][ii][kk]/m_inputScale;
489  }
490 
491 
492  // qNum = |x_t|, only for the missing values (achieved with: 1.0 - m_mask[ii][kk])
493  qNum += clean*clean*(1.0 - m_mask[ii][kk]);
494 
495 
496  // qDen = |x_p - x_t|, only for the missing values
497  qDenom += (clean-noise)*(clean-noise)*(1.0 - m_mask[ii][kk]);
498 
499 
500  // Calculate the number of missing values
501  maskCounter+=static_cast<int>(m_mask[ii][kk]);
502 
503  }
504  }
505  }
506 
507  // MSE = qDen/Number_of_values
508  m_MSError = qDenom/static_cast<double>(maskCounter);
509 
510  //10 log (|x_t|/|x_t-x_p|)
511  m_qMeasure = 10*log10(qNum/qDenom);
512 
513  //Write the error results to a file
514  std::ofstream write_file;
515  write_file.open(path, std::ios_base::out);
516  write_file<<m_MSError<<"\n"<<m_qMeasure<<"\n";
517  write_file.close();
518 
519  return 0.0;
